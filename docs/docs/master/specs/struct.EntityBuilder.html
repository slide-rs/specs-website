pub struct EntityBuilder<'a> {
    pub entity: Entity,
    pub world: &'a World,
}

The entity builder, allowing to build an entity together with its components.

Examples

use specs::{Component, DenseVecStorage, HashMapStorage, World};

struct Health(f32);

impl Component for Health {
    type Storage = HashMapStorage<Self>;
}

struct Pos {
    x: f32,
    y: f32,
}

impl Component for Pos {
    type Storage = DenseVecStorage<Self>;
}

let mut world = World::new();
world.register::<Health>();
world.register::<Pos>();

let entity = world
    .create_entity() // This call returns `EntityBuilder`
    .with(Health(4.0))
    .with(Pos { x: 1.0, y: 3.0 })
    .build(); // Returns the `Entity`
